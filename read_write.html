2.1.2.1 多个Excel 合并导入

Code
combine_sheets <- function(excel_file_path) {
    df <- excel_file_path %>%
        excel_sheets() %>%
        set_names() %>%
        map_df(.f = ~read_excel(path = excel_file_path, sheet = ..1)
               ,.id = "sheet"
        )
    return(df)
}

excel_file_paths <- list.files(path = ".",pattern = ".xlsx",recursive = T)

excel_file_paths
