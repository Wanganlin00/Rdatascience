4.1 逻辑比较

4.1.1 浮点比较

Code
x <- c(1 / 49 * 49, sqrt(2) ^ 2)
x
#> [1] 1 2
x == c(1, 2)
#> [1] FALSE FALSE
print(x, digits = 16)
#> [1] 0.9999999999999999 2.0000000000000004
near(x, c(1, 2))
#> [1] TRUE TRUE

4.1.2 缺失值

Code
# “contagious”
NA == NA
#> [1] NA
is.na(c(TRUE, NA, FALSE))
#> [1] FALSE  TRUE FALSE

4.2 布尔代数

Code
(x1 <- 1:10 %% 2 == 0)
#>  [1] FALSE  TRUE FALSE  TRUE FALSE  TRUE FALSE  TRUE FALSE  TRUE
(y1 <- 1:10 %% 5 == 0)
#>  [1] FALSE FALSE FALSE FALSE  TRUE FALSE FALSE FALSE FALSE  TRUE

(x2 <- which(x1))
#> [1]  2  4  6  8 10
(y2 <- which(y1))
#> [1]  5 10
# 交集
x1 & y1
#>  [1] FALSE FALSE FALSE FALSE FALSE FALSE FALSE FALSE FALSE  TRUE
base::intersect(x2, y2)
#> [1] 10

# 并集
x1 | y1
#>  [1] FALSE  TRUE FALSE  TRUE  TRUE  TRUE FALSE  TRUE FALSE  TRUE
base::union(x2,y2)
#> [1]  2  4  6  8 10  5

# 差集
x1 & !y1
#>  [1] FALSE  TRUE FALSE  TRUE FALSE  TRUE FALSE  TRUE FALSE FALSE
base::setdiff(x2, y2)
#> [1] 2 4 6 8

# 只有一个为真
xor(x1, y1)
#>  [1] FALSE  TRUE FALSE  TRUE  TRUE  TRUE FALSE  TRUE FALSE FALSE
base::setdiff(base::union(x2, y2), base::intersect(x2, y2))
#> [1] 2 4 6 8 5

4.2.1 缺失值

Code
df <- tibble(x = c(TRUE, FALSE, NA))

df |> 
  mutate(
    and = x & NA,  #  A & B   A为TRUE，返回B ；A为FALSE,返回A
    or = x | NA    #  A | B   A为TRUE，返回A ；A为FALSE,返回B
  )
x and or
TRUE NA TRUE
FALSE FALSE NA
NA NA NA

4.4 条件转换

4.4.1 if_else()

Code
x <- c(-3:3, NA)

if_else(x > 0, "+ve", "-ve", missing = "???")
#> [1] "-ve" "-ve" "-ve" "-ve" "+ve" "+ve" "+ve" "???"
if_else(x == 0,
        "0", 
        if_else(x < 0, "-ve", "+ve"), "???")
#> [1] "-ve" "-ve" "-ve" "0"   "+ve" "+ve" "+ve" "???"
Code
x1 <- c(NA, 1, 2, NA)
y1 <- c(3, NA, 4, 6)
if_else(is.na(x1), y1, x1)
#> [1] 3 1 2 6

4.4.2 case_when()

Code
x <- c(-3:3, NA)
case_when(
  x == 0   ~ "0",
  x < 0    ~ "-ve", 
  x > 0    ~ "+ve",
  is.na(x) ~ "???"
)
#> [1] "-ve" "-ve" "-ve" "0"   "+ve" "+ve" "+ve" "???"

## 布尔代数

![](images/boolean_operations.png){fig-align="center" width="75%"}

```{r}
(x1 <- 1:10 %% 2 == 0)
(y1 <- 1:10 %% 5 == 0)

(x2 <- which(x1))
(y2 <- which(y1))
# 交集
x1 & y1
base::intersect(x2, y2)

# 并集
x1 | y1
base::union(x2,y2)

# 差集
x1 & !y1
base::setdiff(x2, y2)

# 只有一个为真
xor(x1, y1)
base::setdiff(base::union(x2, y2), base::intersect(x2, y2))
```
