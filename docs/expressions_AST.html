20.1 抽象语法树

表达式（expressions） 也称为抽象语法树（ abstract syntax trees ，ASTs）。

20.1.1 树结构

  1. 实际运行代码，橙色矩形是分支，紫色符号是函数调用，白色是函数参数

  2. 渲染成书后，灰色矩形是分支，右连函数调用，下连子项参数或函数

Code
lobstr::ast(f(g(1, 2), h(3, 4, i())))
#> █─f 
#> ├─█─g 
#> │ ├─1 
#> │ └─2 
#> └─█─h 
#>   ├─3 
#>   ├─4 
#>   └─█─i

20.1.2 非代码组间

空格和注释

Code
ast(
  f(x,  y)  # important!
)
#> █─f 
#> ├─x 
#> └─y

lobstr::ast(y <-  x)
#> █─`<-` 
#> ├─y 
#> └─x

lobstr::ast(y < -x)
#> █─`<` 
#> ├─y 
#> └─█─`-` 
#>   └─x

20.1.3 中缀调用

Code
x <- 4
`<-`(y,x)
`<-`(y, `*`(x, 10))
expr(`<-`(y, `*`(x, 10)))
#> y <- x * 10
lobstr::ast(y <- x * 10)
#> █─`<-` 
#> ├─y 
#> └─█─`*` 
#>   ├─x 
#>   └─10

20.2 表达式

constant scalars, symbols, call objects, and pairlists

20.2.1 常量

Code
identical(expr(TRUE), TRUE)
#> [1] TRUE
identical(expr(1), 1)
#> [1] TRUE
identical(expr(2L), 2L)
#> [1] TRUE
identical(expr("x"), "x")
#> [1] TRUE

20.2.2 符号

symbol表示对象的名称

Code
# 捕获
expr(x)
#> x

# 字符串转换
rlang::sym("x")
#> x
Code
as_string(expr(x))
#> [1] "x"
str(expr(x))
#>  symbol x
is.symbol(expr(x))
#> [1] TRUE

符号的长度始终为1，多个符号使用rlang::syms()

20.2.3 调用对象

调用对象是一种特殊类型的列表，其中第一个组件指定要调用的函数（通常为符号），其余元素是该函数调用的参数。

Code
lobstr::ast(read.table(file = "important.csv", row.names = FALSE))
#> █─read.table 
#> ├─file = "important.csv" 
#> └─row.names = FALSE

x <- expr(read.table(file = "important.csv", row.names = FALSE))

typeof(x)
#> [1] "language"
Code
x[[1]]
#> read.table
is.symbol(x[[1]])
#> [1] TRUE
as.list(x[-1])
#> $file
#> [1] "important.csv"
#> 
#> $row.names
#> [1] FALSE
x$header <- TRUE
x
#> read.table(file = "important.csv", row.names = FALSE, header = TRUE)

由于 R 灵活的参数匹配规则，从调用中提取特定参数具有挑战性：它可能位于任何位置，具有全名、缩写名称或无名称。若要解决此问题，可以使用 rlang::call_standardise()标准化所有参数来使用全名。

Code
rlang::call_standardise(x)
#> read.table(file = "important.csv", header = TRUE, row.names = FALSE)

函数位置

Code
lobstr::ast("foo"())
#> █─foo
lobstr::ast(foo())
#> █─foo
lobstr::ast(pkg::foo(1))
#> █─█─`::` 
#> │ ├─pkg 
#> │ └─foo 
#> └─1
lobstr::ast(foo(1)(2))
#> █─█─foo 
#> │ └─1 
#> └─2

构造函数调用

Code
call2("mean", x = expr(x), na.rm = TRUE)
#> mean(x = x, na.rm = TRUE)
call2(expr(base::mean), x = expr(x), na.rm = TRUE)
#> base::mean(x = x, na.rm = TRUE)
call2("<-", expr(x), 10)
#> x <- 10

20.2.4 配对列表

Code
f <- expr(function(x, y = 10) x + y)

args <- f[[2]]
args
#> $x
#> 
#> 
#> $y
#> [1] 10
typeof(args)
#> [1] "pairlist"
pl <- pairlist(x = 1, y = 2)
length(pl)
#> [1] 2
pl$x
#> [1] 1

20.2.5 缺失参数

空符号，empty symbol，表示缺失的参数

Code
missing_arg()
typeof(missing_arg())
#> [1] "symbol"
is_missing(missing_arg())
#> [1] TRUE

is_missing(args[[1]])
#> [1] TRUE

...

Code
g <- expr(function(...) list(...))
g
#> function(...) list(...)
args <- g[[2]]
args 
#> $...
is_missing(args[[1]])
#> [1] TRUE

20.2.6 表达式向量

Code
exp1 <- parse(text = c("
x <- 4
x
"))
exp2 <- expression(x <- 4, x)

typeof(exp1)
#> [1] "expression"
typeof(exp2)
#> [1] "expression"


exp1
#> expression(x <- 4, x)

exp2
#> expression(x <- 4, x)

表达式向量的行为也类似于列表

Code
exp1[[1]]
#> x <- 4

20.3 解析表达式

Code
# 字符串
x1 <- "y <- x + 10"
x1
#> [1] "y <- x + 10"
is.call(x1)
#> [1] FALSE

# 表达式
x2 <- rlang::parse_expr(x1)
x2
#> y <- x + 10
is.call(x2)
#> [1] TRUE
Code
x3 <- "a <- 1; a + 1"
rlang::parse_exprs(x3)
#> [[1]]
#> a <- 1
#> 
#> [[2]]
#> a + 1
Code
z <- expr(y <- x + 10)
expr_text(z)
#> [1] "y <- x + 10"
